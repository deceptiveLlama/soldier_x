rule("[GLOBAL] Spawn/Remove Team 1 Roadhog")
{
	event
	{
		Ongoing - Global;
	}

	conditions
	{
		Global.global_init_has_executed == True;
		Global.time_until_roadhog_team_1 > 0;
		Players In Slot(5, Team 1) == Null;
	}

	actions
	{
		disabled Small Message(All Players(All Teams), Custom String("Success! / Started"));
		Wait Until(Global.time_until_roadhog_team_1 == 0, 99999);
		Create Dummy Bot(Hero(Roadhog), Team 1, 5, First Of(Global.roadhog_walk_points_team_1), Direction Towards(First Of(
			Global.roadhog_walk_points_team_1) + Up, Global.roadhog_walk_points_team_1[1] + Up));
		disabled Small Message(All Players(All Teams), Custom String("Created Dummy Bot"));
		Wait Until(Players In Slot(5, Team 1) != Null, 99999);
		"Alive"
		disabled Wait Until(Count Of(Filtered Array(All Players(Team 1), Hero Of(Current Array Element) == Hero(Roadhog) && Is Alive(
			Current Array Element) == True && Is Dummy Bot(Current Array Element) == True)) > 0, 99999);
		Global.roadhog_alive_team_1 = True;
		disabled Small Message(All Players(All Teams), Custom String("Alive"));
		"Dead"
		Wait Until(Is Dead(First Of(Filtered Array(All Players(Team 1), Is Dummy Bot(Current Array Element) == True && Hero Of(
			Current Array Element) == Hero(Roadhog)))), 99999);
		Global.roadhog_alive_team_1 = False;
		disabled Small Message(All Players(All Teams), Custom String("Dead"));
		Global.gold_pile_position_team_1 = Nearest Walkable Position(Position Of(First Of(Filtered Array(All Players(Team 1), Is Dummy Bot(
			Current Array Element) == True && Hero Of(Current Array Element) == Hero(Roadhog)))));
		Global.gold_pile_amount_team_1 = Max(Count Of(Filtered Array(All Players(All Teams), Is Dummy Bot(Current Array Element) == False))
			* 2, 2);
		Global.gold_pile_max_value_team_1 = Count Of(Filtered Array(All Players(All Teams), Is Dummy Bot(Current Array Element) == False))
			* 2;
		Wait(3, Ignore Condition);
		Destroy Dummy Bot(Team 1, 5);
		disabled Small Message(All Players(All Teams), Custom String("Destroyed Dummy Bot"));
		Global.time_until_roadhog_team_1 = Global.upgrade_values_team_1[3];
		Wait Until(Players In Slot(5, Team 1) == Null, 99999);
		disabled Small Message(All Players(All Teams), Custom String("Waiting..."));
		Wait(0.250, Ignore Condition);
		Loop If(Global.max_upgrades_reached_team_1[3] == True);
		Loop If Condition Is True;
	}
}

rule("[PLAYER] Browse Shop")
{
	event
	{
		Ongoing - Each Player;
		All;
		All;
	}

	conditions
	{
		Is Dummy Bot(Event Player) == False;
		Event Player.inside_base == True;
		disabled (Team Of(Event Player) == Team 1 ? Global.time_until_roadhog_team_1 > 0 : Global.time_until_roadhog_team_2 > 0) == True;
		(Is Button Held(Event Player, Button(Primary Fire)) == True || Is Button Held(Event Player, Button(Secondary Fire))
			== True || Is Button Held(Event Player, Button(Interact)) == True) == True;
	}

	actions
	{
		"Primary Fire"
		If(Is Button Held(Event Player, Button(Primary Fire)) == True);
			If(Event Player.selected_upgrade == 0);
				Event Player.selected_upgrade = Count Of(Event Player.shop_upgrades) - 1;
			Else;
				Event Player.selected_upgrade -= 1;
			End;
		"Secondary Fire"
		Else If(Is Button Held(Event Player, Button(Secondary Fire)) == True);
			If(Event Player.selected_upgrade == Count Of(Event Player.shop_upgrades) - 1);
				Event Player.selected_upgrade = 0;
			Else;
				Event Player.selected_upgrade += 1;
			End;
		"Interact"
		Else If(Is Button Held(Event Player, Button(Interact)) == True && Event Player.selected_upgrade > 2);
			"Not enough gold"
			If(Event Player.current_gold < (Team Of(Event Player)
				== Team 1 ? Global.cost_of_upgrades_team_1[Event Player.selected_upgrade] : Global.cost_of_upgrades_team_2[Event Player.selected_upgrade]));
				Start Rule(error_sound, Do Nothing);
				Abort;
			End;
			"Max upgrades reached"
			If((Team Of(Event Player) == Team 1 ? Global.max_upgrades_reached_team_1 : Global.max_upgrades_reached_team_2)
				[Event Player.selected_upgrade] == True);
				Start Rule(error_sound, Do Nothing);
				Abort;
			End;
			Event Player.current_gold -= Team Of(Event Player)
				== Team 1 ? Global.cost_of_upgrades_team_1[Event Player.selected_upgrade] : Global.cost_of_upgrades_team_2[Event Player.selected_upgrade];
			Call Subroutine(buy_upgrade);
			Play Effect(All Players(Team Of(Event Player)), Buff Explosion Sound, Color(White), Team Of(Event Player) == Team 1 ? First Of(
				Filtered Array(All Players(Team 1), Is Dummy Bot(Current Array Element) == True && Hero Of(Current Array Element) == Hero(
				Junkrat))) : First Of(Filtered Array(All Players(Team 2), Is Dummy Bot(Current Array Element) == True && Hero Of(
				Current Array Element) == Hero(Junkrat))), 50);
			Play Effect(All Players(Team Of(Event Player)), Good Pickup Effect, Color(Orange), Team Of(Event Player) == Team 1 ? First Of(
				Filtered Array(All Players(Team 1), Is Dummy Bot(Current Array Element) == True && Hero Of(Current Array Element) == Hero(
				Junkrat))) : First Of(Filtered Array(All Players(Team 2), Is Dummy Bot(Current Array Element) == True && Hero Of(
				Current Array Element) == Hero(Junkrat))), 1);

				Call Subroutine(update_descriptions_team_1);
			Else;
				Call Subroutine(update_descriptions_team_2);
			End;

				Event Player.switching_menu_timer = 0;
				Abort;
			End;
			Stop Chasing Player Variable(Event Player, switching_menu_timer);
			Event Player.switching_menu_timer = 0;
			"Switch Hero"
			If(Event Player.selected_upgrade == 0);
				Set Player Allowed Heroes(Event Player, Remove From Array(Allowed Heroes(Event Player), Hero Of(Event Player)));
				Wait Until(Array Contains(All Heroes, Hero Of(Event Player)) != True, 99999);
				Wait(0.250, Ignore Condition);
				Reset Player Hero Availability(Event Player);
				Abort;
			"Switch Team"
			Else If(Event Player.selected_upgrade == 1);
				If(Number Of Players(Team Of(Event Player)) - Count Of(Filtered Array(All Players(Team Of(Event Player)), Is Dummy Bot(

					All Players(Opposite Team Of(Team Of(Event Player))), Is Dummy Bot(Current Array Element) == True)));

					Abort;

						Current Array Element) == True)) == Number Of Players(Opposite Team Of(Team Of(Event Player))) - Count Of(Filtered Array(
						All Players(Opposite Team Of(Team Of(Event Player))), Is Dummy Bot(Current Array Element) == True)));
					Start Rule(in_queue_to_switch_team, Do Nothing);
					Abort;

						Current Array Element) == True)) > Number Of Players(Opposite Team Of(Team Of(Event Player))) - Count Of(Filtered Array(
						All Players(Opposite Team Of(Team Of(Event Player))), Is Dummy Bot(Current Array Element) == True)));

				End;

			Else If(Event Player.selected_upgrade == 2);
				Start Rule(error_sound, Do Nothing);
				Abort;

rule("- - - - - Subroutine: Switch Team")
{
	event
	{
		Subroutine;
		in_queue_to_switch_team;
	}

	actions
	{
		"You are already in queue"
		If(Event Player.want_to_switch_team == True);
			Abort;
		End;
		"If someone else already want to switch team on your team"
		If(Count Of(Filtered Array(All Players(Team Of(Event Player)), Current Array Element.want_to_switch_team == True)) > 0);
			Start Rule(error_sound, Restart Rule);
			Big Message(Event Player, Custom String("Someone on your team already wants to switch team"));
			Wait(5, Ignore Condition);
			Abort;
		End;
		"If noone wants to switch team on your team"
		If(Count Of(Filtered Array(All Players(Team Of(Event Player)), Current Array Element.want_to_switch_team == True)) == 0);
			Event Player.want_to_switch_team = True;
			If(Count Of(Filtered Array(All Players(Opposite Team Of(Team Of(Event Player))),
				Current Array Element.want_to_switch_team == True)) == 0);
				Big Message(All Players(All Teams), Custom String("{0} wants to switch team. Waiting for someone to switch...", Event Player));
			Else;
				Big Message(All Players(All Teams), Custom String("Switch teams success!", Event Player));
			End;
			"Someone want to switch or failed"
			Wait Until(Count Of(Filtered Array(All Players(Opposite Team Of(Team Of(Event Player))),
				Current Array Element.want_to_switch_team == True)) > 0, 30);
			"Failed"
			If(Count Of(Filtered Array(All Players(Opposite Team Of(Team Of(Event Player))),
				Current Array Element.want_to_switch_team == True)) == 0);
				Big Message(All Players(All Teams), Custom String("Switch team failed for {0}", Event Player));
				Event Player.want_to_switch_team = False;
				Abort;
			End;
			Wait(0.250, Ignore Condition);
			Move Player to Team(Event Player, Opposite Team Of(Team Of(Event Player)), -1);
		End;
	}
}
